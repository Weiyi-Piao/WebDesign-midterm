Example for inheritance 
   class Animal:
    def __init__(self):
        print("An animal is being generated.")
    
    def walk(self):
        print("I am walking......")
           
class Cat(Animal):
    def __init__(self):
        print("I am an cat.")
    def eat(self):
        print("The food is not tasty.")
    def walk(self):
        print("----------------------------------")
        
class Dog(Animal):
    def __init__(self):
        print("")
    def bark(self):
        print("WWWWWWWWWWWWWWWWWWWWW")
